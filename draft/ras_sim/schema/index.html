    🗃️ Schema

    This page includes the complete configuration schema used by the HEC-RAS image to simulate hydraulics.

    🔍 JSON Schema

    {
  "$schema": "http://json-schema.org/draft-07/schema#",
  "$id": "https://example.com/schemas/ras_schema.json",
  "title": "HEC-RAS Execution Schema",
  "version": "0.1.0",
  "description": "Configuration for running a HEC-RAS model using a project file.",
  "type": "object",
  "properties": {
    "program": {
      "type": "string",
      "enum": [
        "ras"
      ]
    },
    "model_directory": {
      "type": "string",
      "minLength": 1
    },
    "model_name": {
      "type": "string",
      "minLength": 1
    }
  },
  "required": [
    "model_directory",
    "model_name"
  ],
  "additionalProperties": false
}
